```c
#include <stdio.h>
int main()
{
    int m = 0,n = 0;
    scanf("%d%d",&m,&n);
    char str[26][26] = {0};
    for(int i = 0;i < m;i ++)
    {
        int temp = 0;
        str[i][0] = 'A' + i;
        for(int j = 1;j < n;j ++)
        {
            if(str[i][j - 1] == 'A')
                temp = 1;
            if(temp)
                str[i][j] = str[i][j - 1] + 1;
            else
                str[i][j] = str[i][j - 1] - 1;
        }
    }
    for(int i = 0;i < m;i ++)
    {
        for(int j = 0;j < n;j ++)
            printf("%c",str[i][j]);
        printf("\n"); 
    }
    return 0;
}
```
