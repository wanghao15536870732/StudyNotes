#include<stdio.h>
#include<string.h>
int main()
{
    char str[50];
    scanf("%s",str);
    for(int i = 0;i < strlen(str);i ++)
    {
        if(65 <= str[i] && str[i] < 90) 
            str[i] += '\1';
        else if(97 <= str[i] && str[i] < 122)
            str[i] += '\1';
        else if(str[i] == 90)
            str[i] += '\7';
        else if(str[i] == 122)  //65 
            str[i] -= '\71';
    }
    printf("%s",str);
    return 0;
}
