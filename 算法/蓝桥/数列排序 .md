```c
#include <stdio.h>

int main()
{
    int n = 0;
    scanf("%d",&n);
    int num[200] = {0};
    for(int j = 0;j < n;j ++)
        scanf("%d",&num[j]);
    for(int i = 1;i < n;i++)
    {
        for(int j = 1;j <= n - i;j ++)
        {
            if(num[j] < num[j - 1])
            {
                int temp = num[j];
                num[j] = num[j - 1];
                num[j - 1] = temp; 
            }
        }
    }
    for(int j = 0;j < n;j ++)
        printf("%d ",num[j]);
    return 0;
}
```
